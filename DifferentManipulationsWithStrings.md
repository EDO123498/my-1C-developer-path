Процедура Check(Команда)
	
	ПроверяемаяСтрока = Объект.Палиндром; 
			
	Если ЗначениеЗаполнено(ПроверяемаяСтрока) Тогда 
		
		СтрокаБезПробелов = НРег(СтрЗаменить(ПроверяемаяСтрока, " ", ""));  
		ПоловинаСтроки = Цел(СтрДлина(ПроверяемаяСтрока)/2);
		ДлинаВсейСтроки = СтрДлина(СтрокаБезПробелов);
		
		ЭтоПалиндром = Истина;
				
		Для Символ = 1 По ПоловинаСтроки  Цикл
			
			СимволСлева   = Сред(СтрокаБезПробелов, Символ, 1);
			СимволСправа  = Сред(СтрокаБезПробелов, ДлинаВсейСтроки - Символ + 1, 1);
			
                        Если СимволСлева <> СимволСправа Тогда 
				ЭтоПалиндром = Ложь;
				Прервать;
			КонецЕсли;
      
		КонецЦикла;     
		
		Сообщить(СтрШаблон("Слово «%1» %2является палиндромом", ПроверяемаяСтрока, ?(ЭтоПалиндром, "", "НЕ ")));	
		
	Иначе
		
		Сообщить("Необходимо ввести слово");
		
	КонецЕсли;	
КонецПроцедуры

&НаКлиенте
Процедура ПоискСамогоДлинногоИСамогоКороткогоСлова (Команда)
	
	Если ЗначениеЗаполнено(Объект.СтрокаПоиск) Тогда
		
		МассивСлов = СтрРазделить(Объект.СтрокаПоиск, " ", Ложь);
		СамоеДлинноеСлово   = "";
		СамоеКороткоеСлово  = МассивСлов[0];
		
		Для Каждого Слово Из МассивСлов Цикл
			
			Если СтрДлина(Слово) > СтрДлина(СамоеДлинноеСлово) Тогда 
				СамоеДлинноеСлово = Слово;
			КонецЕсли;
					
			Если СтрДлина(Слово) < СтрДлина(СамоеКороткоеСлово) Тогда 
				СамоеКороткоеСлово = Слово;
			КонецЕсли;
			
		КонецЦикла;
		
		Сообщить(СтрШаблон("Самое длинное слово «%1» содержит %2 символов, самое короткое слово «%3» содержит %4 символов",
		СамоеДлинноеСлово, СтрДлина(СамоеДлинноеСлово), СамоеКороткоеСлово, СтрДлина(СамоеКороткоеСлово)));
		
	Иначе 
		
		Сообщить("Введите словосочетание или предложение");	
		
	КонецЕсли;
		
КонецПроцедуры
